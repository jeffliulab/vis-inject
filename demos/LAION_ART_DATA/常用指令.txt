========================================
  LAION-Art 数据集下载 — 常用操作指令
========================================

【注意】以下命令均在 HPC 上执行
工作目录: /cluster/tufts/c26sp1ee0141/pliu07/vis_inject/demos/LAION_ART_DATA
数据目录: /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART


================== 1. 提交下载任务 ==================

# 测试下载 (100张图片, 验证环境)
sbatch download_laion_art.sh test

# 全量下载 (~800万张)
sbatch download_laion_art.sh full

# 断点续传 (任务超时后继续，如果设置的是24小时的话，满24小时hpc会杀进程)
sbatch download_laion_art.sh resume


================== 2. 查看任务状态 ==================

# 查看当前所有任务
squeue -u $USER

# 查看任务详情
scontrol show job <JOBID>

# 取消任务
scancel <JOBID>


================== 3. 查看实时日志 ==================

# 查看 SLURM 标准输出 (实时追踪)
tail -f /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/download_<JOBID>.out

# 查看 SLURM 错误输出
tail -f /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/download_<JOBID>.err

# 快速查看最新 SLURM 日志 (自动找最新文件)
ls -t /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/download_*.out | head -1 | xargs tail -50


================== 4. 查看下载结果 ==================

# 查看最新一次运行的完整报告 (末尾50行包含 DOWNLOAD REPORT)
ls -t /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/download_*.out | head -1 | xargs tail -50

# 查看错误统计 (JSON格式, 分类计数)
ls -t /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/error_snapshot_*.json | head -1 | xargs cat

# 查看失败URL列表 (每行一个失败URL及原因)
ls -t /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/failed_urls_*.log | head -1 | xargs head -20

# 查看详细运行日志 (DEBUG级别)
ls -t /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/download_*.log | head -1 | xargs tail -100

# 查看断点续传状态
cat /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/webdataset/.download_state.json

# 查看测试下载的状态
cat /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/webdataset_test/.download_state.json


================== 5. 检查下载数据 ==================

# 已下载的 tar 分片数量和总大小
ls /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/webdataset/*.tar 2>/dev/null | wc -l
du -sh /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/webdataset/

# 运行验证脚本 (检查所有分片完整性)
python verify_dataset.py --data-dir /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/webdataset

# 深度验证 (逐张图片检查, 较慢)
python verify_dataset.py --data-dir /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/webdataset --check-images


================== 6. 磁盘空间 ==================

# 查看存储使用情况
df -h /cluster/tufts/c26sp1ee0141/pliu07/
du -sh /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/
du -sh /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/*/


================== 7. Parquet 元数据 ==================

# 查看已下载的 parquet 分片数 (应为128)
ls /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/metadata/*.parquet | wc -l

# 在 login 节点重新下载 parquet (如需要)
bash download_parquet_metadata.sh


================== 8. 调试技巧 ==================

# 列出所有日志文件 (按时间排序)
ls -lt /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/

# 只看某次运行的所有日志 (替换时间戳)
ls /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/*_20260301_215937.*

# 统计失败原因分布
ls -t /cluster/tufts/c26sp1ee0141/pliu07/LAION_ART/logs/failed_urls_*.log | head -1 | xargs awk -F'|' '{print $1}' | sort | uniq -c | sort -rn

# 查看历史所有 SLURM 任务
sacct -u $USER --format=JobID,JobName,State,ExitCode,Elapsed,Start -S 2026-03-01
