% preamble_extras.tex --- additional packages and macros, layered on
% top of neurips_2026.sty. The style file already loads natbib (via
% \PassOptionsToPackage in main.tex), defines the page geometry, and
% picks Times. We only add what the body actually uses.

% --- Encoding and microtypography (NeurIPS shell already shows utf8/T1) -
\usepackage[utf8]{inputenc}
\usepackage[T1]{fontenc}
\usepackage{microtype}

% --- Hyperlinks + URL typesetting -------------------------------------
\usepackage{hyperref}
\usepackage{url}
\hypersetup{
  colorlinks=true,
  linkcolor=black!70,
  citecolor=black!70,
  urlcolor=black!70,
  pdftitle={VisInject: Disruption Not Equal Injection -- A Dual-Dimension Evaluation of Universal Adversarial Attacks on Vision-Language Models},
  pdfauthor={Pang Liu, Yingjie Lao},
}

% --- Graphics ---------------------------------------------------------
% Reuse the figures already shipped with the xarchive paper rather than
% duplicating them here. This means the present file does NOT introduce
% new image assets; it points at the existing ones.
\usepackage{graphicx}
% Three search roots, in order:
%   1. local figures/ (just the pipeline.tex forwarder lives here),
%   2. xarchive-report/paper/figures/ (for unprefixed includes like
%      ORIGIN_dog.png, clean_ORIGIN_bill.png, HF-downloads.png, ...),
%   3. xarchive-report/paper/ (for explicit "figures/per_vlm.pdf" and
%      "figures/triptych.pdf" references in sections 06 and 07).
\graphicspath{{figures/}{../../xarchive-report/paper/figures/}{../../xarchive-report/paper/}}
